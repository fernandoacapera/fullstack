function calcularMediaAprovados(listaDeNotas) {
    const aprovados = listaDeNotas.filter(nota => nota >= 5);
    const soma = aprovados.reduce((acumulador, nota) => acumulador + nota, 0);
    return soma / aprovados.length;
}
